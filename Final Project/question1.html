def floyd_warshall_with_path(W, start, end):
    n = len(W) 
    D =[]
    for i in range(n):
        col = []
        for j in range(n):
            col.append(W[i][j])
        D.append(col)
    P = []
    for i in range(n):
        col = []
        for j in range(n):
            col.append(None)
        P.append(col)
    for k in range(n):
        for i in range(n):
            for j in range(n):
                if D[i][j] > D[i][k] + D[k][j]:
                    P[i][j] = k
                    D[i][j] = D[i][k] + D[k][j]
                    
    path = []
    if D[start][end] != float('inf'):
        path.append(start)
        build_path(P, start, end, path)
        path.append(end)
    return path

def build_path(P, start, end, path):
    if P[start][end] is not None:
        build_path(P, start, P[start][end], path)
        path.append(P[start][end])
        build_path(P, P[start][end], end, path)
